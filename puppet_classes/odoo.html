class odoo (
  $config_file         = '/etc/odoo/odoo.conf',
  $install_wkhtmltopdf = false,
  $settings            = {},
  $version             = present,
  ) {
  if $install_wkhtmltopdf {
    package { 'wkhtmltopdf':
      ensure => present,
      notify => Service['odoo']
    }
  }

  package { 'odoo':
    ensure => $version,
    notify => Service['odoo']
  }

  if $::osfamily == 'RedHat' {
    exec { '/usr/bin/systemctl daemon-reload':
      refreshonly => true,
      subscribe   => Package['odoo'],
    }
  }

  $defaults = {
    path    => $config_file,
    require => Package['odoo'],
    notify  => Service['odoo'],
  }

  create_ini_settings($settings, $defaults)

  service { 'odoo':
    ensure => running,
    enable => true,
  }
}
